package com.falardeau.tools;

public class TableAlgorithms {

    public void reverse(int[] tab){

        int len = tab.length - 1;
        if(len < 2) return;

        int i = 0, j = len;

        while(i < j){

            int temp = tab[i];
            tab[i] = tab[j];
            tab[j] = temp;

            i++; j--;
        }
    }

    public void print(int[] tab){
        for(int i = 0; i < tab.length; i++){
            System.out.print(tab[i] + " ");
        }
        System.out.println();
    }
}
